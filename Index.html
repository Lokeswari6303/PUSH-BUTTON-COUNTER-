Arduino Temperature Sensor Project

This project demonstrates how to read temperature using an LM35 sensor and display the result on a 16x2 LCD or the Serial Monitor using Arduino.

// Include the LCD library
#include <LiquidCrystal.h>

// Define sensor and LCD pins
const int sensorPin = A0;
LiquidCrystal lcd(12, 11, 5, 4, 3, 2);

void setup() {
  lcd.begin(16, 2);        // Start the LCD
  Serial.begin(9600);      // Start the Serial Monitor
  lcd.print("Temp: ");
}

void loop() {
  int sensorValue = analogRead(sensorPin);
  float voltage = sensorValue * (5.0 / 1023.0);
  float temperatureC = voltage * 100;

  // Display on LCD
  lcd.setCursor(6, 0);
  lcd.print(temperatureC);
  lcd.print(" C ");

  // Display on Serial Monitor
  Serial.print("Temperature: ");
  Serial.print(temperatureC);
  Serial.println(" Â°C");

  delay(1000); // Wait 1 second
}
